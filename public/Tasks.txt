with temp as (
  select subscriber_id, dates from (    
    select subscriber_id, min(date(_metadata_timestamp)) min_date, max(date(_metadata_timestamp)) max_date
     from `excitel-test.test.subscriber_log` where subscriber_id=75156175
    group by subscriber_id
  ), unnest(generate_date_array(min_date, max_date)) dates
)
select b.subscriber_id, dates, 
  first_value(status ignore nulls) over win as status
from temp a
left join  `excitel-test.test.subscriber_log` b 
on b.subscriber_id=a.subscriber_id and date(b._metadata_timestamp)=a.dates
order by dates

WITH ranked AS (
  SELECT *, ROW_NUMBER() OVER (PARTITION BY subscriber_id ORDER BY _metadata_timestamp DESC) AS rn
  FROM `excitel-test.test.subscriber_log` where subscriber_id between 75156175 and 75159275 
)
SELECT * FROM ranked where rn = 1

with qry as (SELECT * ,
  LAG(status)
    OVER (PARTITION BY subscriber_id ORDER BY _metadata_timestamp ASC) AS preceding_status,
    FROM `excitel-test.test.subscriber_log` where subscriber_id between 75156175 and 75159275)

select CASE WHEN end_stamp is null THEN true ELSE false END AS is_current, * from (select  _metadata_timestamp as start_stamp,    
      LEAD(_metadata_timestamp) 
    OVER (PARTITION BY subscriber_id ORDER BY _metadata_timestamp ASC) AS end_stamp,
    * from qry
    where status<>preceding_status or preceding_status is null
)
order by subscriber_id,start_stamp
